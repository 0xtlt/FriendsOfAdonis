---
title: Security
---

You can define which security mechanisms should be used for your operations using the `@ApiSecurity` decorator.

```ts
@ApiSecurity("basic")
export default class UsersController {}
```

The mechanisms must be define on your document, you can do this directly from your `config/openapi.ts` configuration file.

```ts title="config/openapi.ts"
import { defineConfig } from "@foadonis/openapi";

export default defineConfig({
  document: {
    components: {
      securitySchemes: {
        basic: {
          type: "http",
          scheme: "basic",
        },
      },
    },
  },
});
```

## Examples

### Basic authentication

```ts
@ApiBasicAuth()
export default class UsersController {}
```

```ts title="config/openapi.ts"
import { defineConfig } from "@foadonis/openapi";

export default defineConfig({
  document: {
    components: {
      securitySchemes: {
        basic: {
          type: "http",
          scheme: "basic",
        },
      },
    },
  },
});
```

### Bearer authentication

```ts
@ApiBearerAuth()
export default class UsersController {}
```

```ts title="config/openapi.ts"
import { defineConfig } from "@foadonis/openapi";

export default defineConfig({
  document: {
    components: {
      securitySchemes: {
        bearer: {
          type: "http",
          scheme: "bearer",
          bearerFormat: "JWT",
        },
      },
    },
  },
});
```

### OAuth2 authentication

```ts
@ApiOAuth2("users:read", "global:read")
export default class UsersController {}
```

```ts title="config/openapi.ts"
import { defineConfig } from "@foadonis/openapi";

export default defineConfig({
  document: {
    components: {
      securitySchemes: {
        bearer: {
          type: "oauth2",
          flows: {
            implicit: {
              authorizationUrl: "https://example.com/api/oauth/dialog",
              scopes: {
                "users:read": "read users",
                "global:read": "global read",
              },
            },
          },
        },
      },
    },
  },
});
```
